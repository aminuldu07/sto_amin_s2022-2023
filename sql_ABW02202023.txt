--BWANIMAL_02202023__BW1 (terminal bw SELECTION) 
SELECT DISTINCT BW.STUDYID, BW.USUBJID,BW.BWTESTCD,BW.BWSTRESN, BW.BWSTRESU, EX.EXROUTE, EX.EXTRT, EX.EXDOSE, EX.EXDOSU, DM.SEX, DM.ARMCD, DM.SETCD, DM.ARM AS DMARM
FROM (BW INNER JOIN EX ON BW.STUDYID = EX.STUDYID)
INNER JOIN DM ON DM.USUBJID=BW.USUBJID
WHERE BW.BWTESTCD == "TERMBW";
