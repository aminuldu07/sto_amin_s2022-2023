SELECT DISTINCT BW.STUDYID, BW.USUBJID, BW.BWTESTCD, BW.BWSTRESN,BW.BWSTRESU,EX.EXROUTE, EX.EXTRT, EX.EXDOSE, EX.EXDOSU, TX.STUDYID AS TXSTUDYID,TX.TXPARMCD, TX.TXVAL, TX.SETCD, DM.STUDYID AS DMSTUDY, DM.USUBJID AS DMUSID, DM.SETCD AS DMSETCD,DM.ARM AS DMARM
FROM ((DM INNER JOIN TX ON DM.SETCD = TX.SETCD)
INNER JOIN BW ON BW.USUBJID=DM.USUBJID)
INNER JOIN EX ON BW.USUBJID = EX.USUBJID
WHERE TX.TXPARMCD == "TCNTRL" AND BW.BWTESTCD == "TERMBW";
