SELECT DISTINCT BW.STUDYID, DM.SETCD, TX."SET",EX.EXROUTE,TE.TEDUR,TX.TXPARMCD, TX.TXVAL, BW.USUBJID, BW.VISITDY, BW.BWTESTCD, BW.BWSTRESN, BW.BWSTRESU, DM.SEX, TS.TSVAL AS "SPECIES"
FROM ((((BW INNER JOIN DM ON DM.USUBJID=BW.USUBJID)
INNER JOIN TS ON TS.STUDYID=BW.STUDYID)
INNER JOIN TE ON TE.STUDYID=BW.STUDYID)
INNER JOIN EX ON EX.STUDYID=BW.STUDYID)
INNER JOIN TX ON TX.STUDYID=BW.STUDYID
WHERE TE.TEDUR IS NOT '' AND TE.TEDUR IS NOT NULL AND  TS.TSPARMCD LIKE "%SPECIES%"AND TX.TXPARMCD == "TRTDOS"AND TX.SETCD == DM.SETCD AND BW.BWTESTCD == "TERMBW" AND TS.TSVAL == "RAT";
